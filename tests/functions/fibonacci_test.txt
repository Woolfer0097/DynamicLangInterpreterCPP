// Test: Fibonacci Recursive Function
// Tests recursive function calls and closures

test fibonacci_recursive() is
    var fib := func(n) is
        if n <= 1 then
            return n
        else
            return fib(n - 1) + fib(n - 2)
        end
    end

    assert fib(0) == 0
    assert fib(1) == 1
    assert fib(2) == 1
    assert fib(3) == 2
    assert fib(4) == 3
    assert fib(5) == 5
    assert fib(6) == 8
    assert fib(7) == 13
    assert fib(10) == 55
    print fib(20)
end

fibonacci_recursive()
print "Fibonacci test passed!"
