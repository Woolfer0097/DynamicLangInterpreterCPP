// Demo: Named Tuples
// Showcases tuple creation with named and unnamed fields

print "=== Named Tuples Demo ==="

// Named tuple (like a struct)
var person := {name := "Alice", age := 30, city := "New York"}
print "Person record:"
print "  Name:", person.name
print "  Age:", person.age
print "  City:", person.city

// Unnamed tuple (positional access)
var point := {10, 20, 30}
print "3D Point:"
print "  X:", point.1
print "  Y:", point.2
print "  Z:", point.3

// Mixed named and unnamed fields
var data := {id := 1, 3.14, active := true, "extra"}
print "Mixed tuple:"
print "  id:", data.id, "or", data.1
print "  value:", data.2
print "  active:", data.active, "or", data.3
print "  extra:", data.4

// Tuple modification
person.age := 31
print "Updated age:", person.age

// This would combine two tuples into one

print "=== Demo Complete ==="
