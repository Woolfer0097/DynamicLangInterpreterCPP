// Test: Literals and Basic Types
// Tests all literal types and their representations

test integer_literals() is
    var i1 := 0
    var i2 := 42
    var i3 := -123
    assert i1 == 0
    assert i2 == 42
    assert i3 == -123
    assert i1 is int
    assert i2 is int
    assert i3 is int
end

test real_literals() is
    var r1 := 3.14
    var r2 := -2.5
    var r3 := 0.0
    var r4 := 100.0
    assert r1 == 3.14
    assert r2 == -2.5
    assert r3 == 0.0
    assert r4 == 100.0
    assert r1 is real
    assert r2 is real
    assert r3 is real
    assert r4 is real
end

test boolean_literals() is
    var b1 := true
    var b2 := false
    assert b1 == true
    assert b2 == false
    assert b1 is bool
    assert b2 is bool
end

test string_literals() is
    var s1 := "hello"
    var s2 := 'hello world'
    var s3 := ""
    var s4 := "123"
    assert s1 == "hello"
    assert s2 == "hello world"
    assert s3 == ""
    assert s4 == "123"
    assert s1 is string
    assert s2 is string
    assert s3 is string
    assert s4 is string
end

// Execute all tests
print "Running literals tests..."
integer_literals()
real_literals()
boolean_literals()
string_literals()
print "All literals tests passed!"
